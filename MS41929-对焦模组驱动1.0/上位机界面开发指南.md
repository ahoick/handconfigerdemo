## 2. 网络摄像头视频流读取

### 2.1 基本原理
网络摄像头通常通过RTSP协议或HTTP流提供视频数据。我们可以使用OpenCV库来捕获和处理这些视频流。

### 2.2 实现代码
```python
import cv2

class CameraHandler:
    def __init__(self, camera_url):
        """初始化摄像头处理器
        Args:
            camera_url: 摄像头RTSP地址或本地设备ID
        """
        self.camera_url = camera_url
        self.cap = None
    
    def open_camera(self):
        """打开摄像头连接
        Returns:
            bool: 连接是否成功
        """
        try:
            self.cap = cv2.VideoCapture(self.camera_url)
            if not self.cap.isOpened():
                print("无法打开摄像头")
                return False
            return True
        except Exception as e:
            print(f"打开摄像头失败: {e}")
            return False
    
    def get_frame(self):
        """获取一帧图像
        Returns:
            numpy.ndarray: 图像帧，失败返回None
        """
        if not self.cap or not self.cap.isOpened():
            return None
        
        ret, frame = self.cap.read()
        if not ret:
            print("无法获取图像帧")
            return None
        
        return frame
    
    def release(self):
        """释放摄像头资源"""
        if self.cap:
            self.cap.release()

# 示例使用
if __name__ == "__main__":
    # 网络摄像头RTSP地址示例
    camera_url = "rtsp://admin:password@192.168.1.100:554/stream1"
    # 本地摄像头
    # camera_url = 0
    
    camera = CameraHandler(camera_url)
    if camera.open_camera():
        while True:
            frame = camera.get_frame()
            if frame is not None:
                cv2.imshow("Camera", frame)
                
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        
        camera.release()
        cv2.destroyAllWindows()
```

## 3. 串口通信实现

### 3.1 基本原理
使用 pyserial 库与 STM32 开发板进行串口通信。采用**单字节指令 + 单字节响应**的简约协议，无帧头/帧尾/校验字节。

### 3.2 协议说明

#### 指令集

| 指令 | 说明 |
|------|------|
| `0x00` | 心跳包，刷新看门狗 |
| `0x01` ~ `0x04` | AB电机 正转/反转/停止/刹车 |
| `0x05` ~ `0x08` | CD电机 正转/反转/停止/刹车 |
| `0x10` ~ `0x73` | 速度设置（对应 1~100，指令 = 速度 + 0x0F） |

#### 响应码

| 响应 | 含义 |
|------|------|
| `0x55` | 执行成功 |
| `0xEE` | 无效指令 |
| `0x99` | 看门狗超时报警（下位机主动上报） |

### 3.3 实现代码
```python
import serial

# 单字节指令定义
CMD_HEARTBEAT   = 0x00  # 心跳包
CMD_AB_FORWARD  = 0x01  # AB电机正转
CMD_AB_REVERSE  = 0x02  # AB电机反转
CMD_AB_STOP     = 0x03  # AB电机停止
CMD_AB_BRAKE    = 0x04  # AB电机刹车
CMD_CD_FORWARD  = 0x05  # CD电机正转
CMD_CD_REVERSE  = 0x06  # CD电机反转
CMD_CD_STOP     = 0x07  # CD电机停止
CMD_CD_BRAKE    = 0x08  # CD电机刹车

ACK_OK  = 0x55
ACK_ERR = 0xEE

def speed_cmd(speed: int) -> int:
    """速度值(1~100) → 指令字节(0x10~0x73)"""
    return speed + 0x0F

class SerialCommunicator:
    def __init__(self, port, baudrate=115200, timeout=0.5):
        self.port = port
        self.baudrate = baudrate
        self.timeout = timeout
        self.ser = None
    
    def open_port(self):
        try:
            self.ser = serial.Serial(
                port=self.port,
                baudrate=self.baudrate,
                timeout=self.timeout
            )
            if self.ser.is_open:
                print(f"串口 {self.port} 打开成功")
                return True
            return False
        except Exception as e:
            print(f"打开串口失败: {e}")
            return False
    
    def send_byte(self, cmd_byte):
        """发送单字节指令，返回响应"""
        if not self.ser or not self.ser.is_open:
            print("串口未打开")
            return None
        try:
            self.ser.write(bytes([cmd_byte]))
            response = self.ser.read(1)
            if len(response) == 1:
                return response[0]
            return None
        except Exception as e:
            print(f"发送失败: {e}")
            return None
    
    def send_command(self, cmd_byte, desc=""):
        """发送指令并验证响应"""
        resp = self.send_byte(cmd_byte)
        if resp == ACK_OK:
            print(f"{desc} → 成功 (0x{resp:02X})")
            return True
        elif resp == ACK_ERR:
            print(f"{desc} → 无效指令 (0x{resp:02X})")
            return False
        else:
            print(f"{desc} → 无响应")
            return False
    
    def close_port(self):
        if self.ser and self.ser.is_open:
            self.ser.close()
            print(f"串口 {self.port} 已关闭")

# 示例使用
if __name__ == "__main__":
    ser = SerialCommunicator('COM3')
    if ser.open_port():
        # 设置速度 100 → 指令 0x73
        ser.send_command(speed_cmd(100), "设置速度 100")
        # AB 正转
        ser.send_command(CMD_AB_FORWARD, "AB正转")
        # 心跳保活
        ser.send_command(CMD_HEARTBEAT, "心跳")
        # AB 停止
        ser.send_command(CMD_AB_STOP, "AB停止")
        ser.close_port()
```

## 5. 界面设计与实现

### 5.1 PyQt5界面示例
```python
import sys
from PyQt5.QtWidgets import (QApplication, QMainWindow, QWidget, QVBoxLayout, 
                            QHBoxLayout, QPushButton, QSlider, QLabel, 
                            QComboBox, QTextEdit, QGroupBox, QGridLayout)
from PyQt5.QtCore import Qt, QThread, pyqtSignal
from PyQt5.QtGui import QImage, QPixmap
import cv2
import time

class CameraThread(QThread):
    """摄像头线程"""
    frame_signal = pyqtSignal(QImage)
    
    def __init__(self, camera_url):
        super().__init__()
        self.camera_url = camera_url
        self.running = False
    
    def run(self):
        self.running = True
        cap = cv2.VideoCapture(self.camera_url)
        
        while self.running:
            ret, frame = cap.read()
            if ret:
                # 转换为RGB格式
                rgb_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                # 转换为QImage
                h, w, ch = rgb_frame.shape
                bytes_per_line = ch * w
                q_image = QImage(rgb_frame.data, w, h, bytes_per_line, QImage.Format_RGB888)
                self.frame_signal.emit(q_image)
            time.sleep(0.03)  # 控制帧率
        
        cap.release()
    
    def stop(self):
        self.running = False
        self.wait()

class MotorControlApp(QMainWindow):
    """电机控制应用"""
    def __init__(self):
        super().__init__()
        self.init_ui()
        self.serial_comm = None
        self.camera_thread = None
    
    def init_ui(self):
        """初始化界面"""
        self.setWindowTitle("电机控制上位机")
        self.setGeometry(100, 100, 800, 600)
        
        # 主布局
        central_widget = QWidget()
        main_layout = QVBoxLayout(central_widget)
        
        # 视频显示区域
        video_group = QGroupBox("摄像头监控")
        video_layout = QVBoxLayout()
        self.video_label = QLabel("视频流将显示在这里")
        self.video_label.setAlignment(Qt.AlignCenter)
        self.video_label.setFixedSize(640, 480)
        video_layout.addWidget(self.video_label)
        video_group.setLayout(video_layout)
        main_layout.addWidget(video_group)
        
        # 控制区域
        control_group = QGroupBox("电机控制")
        control_layout = QGridLayout()
        
        # 串口选择
        control_layout.addWidget(QLabel("串口:"), 0, 0)
        self.port_combo = QComboBox()
        self.port_combo.addItems(["COM1", "COM2", "COM3", "COM4", "COM5"])
        control_layout.addWidget(self.port_combo, 0, 1)
        
        # 连接按钮
        self.connect_btn = QPushButton("连接")
        self.connect_btn.clicked.connect(self.toggle_connection)
        control_layout.addWidget(self.connect_btn, 0, 2)
        
        # 电机通道
        control_layout.addWidget(QLabel("电机通道:"), 1, 0)
        self.channel_combo = QComboBox()
        self.channel_combo.addItems(["通道0 (AB)", "通道1 (CD)"])
        control_layout.addWidget(self.channel_combo, 1, 1, 1, 2)
        
        # 控制按钮
        btn_layout = QHBoxLayout()
        self.forward_btn = QPushButton("正转")
        self.forward_btn.clicked.connect(lambda: self.control_motor(MOTOR_FORWARD))
        btn_layout.addWidget(self.forward_btn)
        
        self.reverse_btn = QPushButton("反转")
        self.reverse_btn.clicked.connect(lambda: self.control_motor(MOTOR_REVERSE))
        btn_layout.addWidget(self.reverse_btn)
        
        self.stop_btn = QPushButton("停止")
        self.stop_btn.clicked.connect(lambda: self.control_motor(MOTOR_STOP))
        btn_layout.addWidget(self.stop_btn)
        
        self.brake_btn = QPushButton("刹车")
        self.brake_btn.clicked.connect(lambda: self.control_motor(MOTOR_BRAKE))
        btn_layout.addWidget(self.brake_btn)
        
        control_layout.addLayout(btn_layout, 2, 0, 1, 3)
        
        # 速度控制
        control_layout.addWidget(QLabel("速度:"), 3, 0)
        self.speed_slider = QSlider(Qt.Horizontal)
        self.speed_slider.setRange(1, 255)
        self.speed_slider.setValue(100)
        control_layout.addWidget(self.speed_slider, 3, 1)
        
        self.speed_label = QLabel("100")
        control_layout.addWidget(self.speed_label, 3, 2)
        self.speed_slider.valueChanged.connect(lambda value: self.speed_label.setText(str(value)))
        
        # 速度设置按钮
        self.set_speed_btn = QPushButton("设置速度")
        self.set_speed_btn.clicked.connect(self.set_speed)
        control_layout.addWidget(self.set_speed_btn, 4, 0, 1, 3)
        
        # 状态查询按钮
        self.query_btn = QPushButton("查询状态")
        self.query_btn.clicked.connect(self.query_status)
        control_layout.addWidget(self.query_btn, 5, 0, 1, 3)
        
        # 日志区域
        self.log_text = QTextEdit()
        self.log_text.setReadOnly(True)
        self.log_text.setFixedHeight(100)
        control_layout.addWidget(QLabel("日志:"), 6, 0)
        control_layout.addWidget(self.log_text, 7, 0, 1, 3)
        
        control_group.setLayout(control_layout)
        main_layout.addWidget(control_group)
        
        self.setCentralWidget(central_widget)
    
    def toggle_connection(self):
        """切换串口连接状态"""
        if self.serial_comm and self.serial_comm.ser and self.serial_comm.ser.is_open:
            # 关闭连接
            self.serial_comm.close_port()
            self.connect_btn.setText("连接")
            self.log("串口已关闭")
        else:
            # 打开连接
            port = self.port_combo.currentText()
            self.serial_comm = SerialCommunicator(port)
            if self.serial_comm.open_port():
                self.connect_btn.setText("断开")
                self.log(f"串口 {port} 连接成功")
                # 启动摄像头
                self.start_camera()
            else:
                self.log(f"串口 {port} 连接失败")
    
    def start_camera(self):
        """启动摄像头"""
        # 这里使用本地摄像头，实际使用时替换为网络摄像头地址
        camera_url = 0  # 本地摄像头
        # camera_url = "rtsp://admin:password@192.168.1.100:554/stream1"
        
        self.camera_thread = CameraThread(camera_url)
        self.camera_thread.frame_signal.connect(self.update_video)
        self.camera_thread.start()
    
    def update_video(self, q_image):
        """更新视频显示"""
        self.video_label.setPixmap(QPixmap.fromImage(q_image))
    
    def control_motor(self, action):
        """控制电机"""
        if not self.serial_comm or not self.serial_comm.ser or not self.serial_comm.ser.is_open:
            self.log("请先连接串口")
            return
        
        channel = self.channel_combo.currentIndex()
        speed = self.speed_slider.value()
        
        actions = {
            MOTOR_FORWARD: "正转",
            MOTOR_REVERSE: "反转",
            MOTOR_STOP: "停止",
            MOTOR_BRAKE: "刹车"
        }
        
        success = self.serial_comm.send_command(CMD_MOTOR_CONTROL, action, channel, speed)
        if success:
            self.log(f"电机{actions[action]}命令发送成功")
        else:
            self.log(f"电机{actions[action]}命令发送失败")
    
    def set_speed(self):
        """设置速度"""
        if not self.serial_comm or not self.serial_comm.ser or not self.serial_comm.ser.is_open:
            self.log("请先连接串口")
            return
        
        channel = self.channel_combo.currentIndex()
        speed = self.speed_slider.value()
        
        success = self.serial_comm.send_command(CMD_SPEED_SET, 0, channel, speed)
        if success:
            self.log(f"速度设置为 {speed} 成功")
        else:
            self.log("速度设置失败")
    
    def query_status(self):
        """查询状态"""
        if not self.serial_comm or not self.serial_comm.ser or not self.serial_comm.ser.is_open:
            self.log("请先连接串口")
            return
        
        channel = self.channel_combo.currentIndex()
        success = self.serial_comm.send_command(CMD_STATUS_QUERY, 0, channel)
        if success:
            self.log("状态查询命令发送成功")
        else:
            self.log("状态查询命令发送失败")
    
    def log(self, message):
        """添加日志"""
        self.log_text.append(f"[{time.strftime('%H:%M:%S')}] {message}")
        # 自动滚动到底部
        self.log_text.verticalScrollBar().setValue(self.log_text.verticalScrollBar().maximum())
    
    def closeEvent(self, event):
        """关闭事件"""
        if self.camera_thread:
            self.camera_thread.stop()
        if self.serial_comm and self.serial_comm.ser and self.serial_comm.ser.is_open:
            self.serial_comm.close_port()
        event.accept()

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = MotorControlApp()
    window.show()
    sys.exit(app.exec_())
```

| CMD_CD_REVERSE | `0x06` | CD电机反转 |
| CMD_CD_STOP | `0x07` | CD电机停止（滑行） |
| CMD_CD_BRAKE | `0x08` | CD电机刹车（抱死） |
| 速度 1~100 | `0x10`~`0x73` | 速度 = 指令 - 0x0F |

| 响应码 | 值 | 含义 |
|-------|----|------|
| ACK_OK | `0x55` | 执行成功 |
